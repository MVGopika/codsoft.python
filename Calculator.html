def add(num1,num2): return num1+num2 def subtract(num1,num2): return num1-num2 def divide(num1,num2): return num1/num2 def multiply(num1,num2): return num1*num2 def modulus(num1,num2): return num1%num2 def square(num1): return num1**2 def cube(num1): return num1**3 def root(num1): return num1**(1/2) print("Select the operation for the calculation:\n"\ "1.Add the number\n"\ "2.Subtract the number\n"\ "3.Divide the number\n"\ "4.Multiply the number\n"\ "5.The modulus of the number\n"\ "6.The square of the number\n"\ "7.Cube of the number\n"\ "8.Root of the number\n") select=int(input("Select operations from the above 1,2,3,4,5,6,7,8:")) number_1=int(input("Enter the first number:")) number_2=int(input("Enter the second number:")) if select==1: print(number_1,"+",number_2,"=",add(number_1,number_2)) elif select==2: print(number_1,"-",number_2,"=",subtract(number_1,number_2)) elif select==3: print(number_1,"/",number_2,"=",divide(number_1,number_2)) elif select==4: print(number_1,"*",number_2,"=",multiply(number_1,number_2)) elif select==5: print(number_1,"%",number_2,"=",modulus(number_1,number_2)) elif select==6: print(number_1,"**2=",square(number_1)) elif select==7: print(number_1,"**3=",cube(number_1)) elif select==8: print(number_1,"**(1/2)",root(number_1)) else: print("The operation is not found in the above numbers.")
